025package com.github.siroshun09.event4j.handlerlist;
026
027import com.github.siroshun09.event4j.event.Event;
028import com.github.siroshun09.event4j.listener.Listener;
029import org.jetbrains.annotations.Contract;
030import org.jetbrains.annotations.NotNull;
031import org.jetbrains.annotations.Nullable;
032
033import java.util.logging.Logger;
034
035/**
036 * An interface that manages listeners of the event.
037 *
038 * @param <T> the type of event
039 */
040public interface HandlerList<T extends Event> {
041
042    /**
043     * Creates a new {@link HandlerList}.
044     *
045     * @param <T> the type of event
046     * @return a new {@link HandlerList}
047     */
048    @Contract(value = " -> new", pure = true)
049    @NotNull
050    static <T extends Event> HandlerList<T> create() {
051        return new HandlerListImpl<>();
052    }
053
054    /**
055     * Dispatches the event to subscribed listeners.
056     * <p>
057     * If the exception logger has been set by {@link HandlerList#setExceptionLogger(Logger)},
058     * it should log any exceptions that occur while processing this method.
059     *
060     * @param event the event instance
061     */
062    void post(@NotNull T event);
063
064    /**
065     * Subscribes the listener.
066     * <p>
067     * This method subscribes the listener with {@link Priority#NORMAL}.
068     *
069     * @param key      the key to be used to subscribe the listener
070     * @param listener the listener to subscribe
071     * @return {@code true} if it is subscribed, {@code false} otherwise
072     */
073    boolean subscribe(@NotNull Key key, @NotNull Listener<T> listener);
074
075    /**
076     * Subscribes the listener.
077     *
078     * @param key      the key to be used to subscribe the listener
079     * @param listener the listener to subscribe
080     * @param priority the priority of the listener
081     * @return {@code true} if it is subscribed, {@code false} otherwise
082     */
083    boolean subscribe(@NotNull Key key, @NotNull Listener<T> listener, @NotNull Priority priority);
084
085    /**
086     * Unsubscribes the listener.
087     * <p>
088     * The listener instance must be the same as when you subscribed it to be unregistered.
089     *
090     * @param listener the listener instance to unsubscribe
091     * @return {@code true} if it is unsubscribed, {@code false} otherwise
092     */
093    boolean unsubscribe(@NotNull Listener<T> listener);
094
095    /**
096     * Unsubscribes the listener of the specified priority.
097     * <p>
098     * The listener instance must be the same as when you subscribed it to be unsubscribed.
099     *
100     * @param listener the listener instance to unsubscribe
101     * @param priority the priority of the listener.
102     * @return {@code true} if it is unsubscribed, {@code false} otherwise
103     */
104    boolean unsubscribe(@NotNull Listener<T> listener, @NotNull Priority priority);
105
106    /**
107     * Unsubscribes all listeners subscribed with the specified {@link Key}.
108     *
109     * @param key the key to unsubscribe
110     */
111    void unsubscribeAll(@NotNull Key key);
112
113    /**
114     * Gets the number of subscribed listeners.
115     *
116     * @return the number of subscribed listeners.
117     */
118    int getSize();
119
120    /**
121     * Sets the logger to logging exceptions.
122     * <p>
123     * The default exception logger is null, which means that
124     * it will not log any exceptions that occur while processing an event.
125     *
126     * @param logger the logger to logging exceptions
127     */
128    void setExceptionLogger(@Nullable Logger logger);
129}
