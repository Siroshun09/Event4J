025package com.github.siroshun09.event4j.bus;
026
027import com.github.siroshun09.event4j.event.Event;
028import com.github.siroshun09.event4j.handlerlist.HandlerList;
029import com.github.siroshun09.event4j.handlerlist.Key;
030import org.jetbrains.annotations.NotNull;
031import org.jetbrains.annotations.Nullable;
032
033import java.util.HashMap;
034import java.util.Map;
035import java.util.Objects;
036
037/**
038 * A class that implements {@link EventBus}
039 */
040public class SimpleEventBus implements EventBus {
041
042    private final Map<Class<?>, HandlerList<?>> handlers = new HashMap<>();
043
044    /**
045     * Creates a {@link SimpleEventBus}.
046     */
047    public SimpleEventBus() {
048    }
049
050    /**
051     * {@inheritDoc}
052     */
053    @NotNull
054    @Override
055    public <T extends Event> HandlerList<T> getHandlerList(@NotNull Class<T> eventClass) {
056        Objects.requireNonNull(eventClass);
057
058        HandlerList<T> handlerList = searchHandlerList(eventClass);
059
060        if (handlerList == null) {
061            handlerList = HandlerList.create();
062            handlers.put(eventClass, handlerList);
063        }
064
065        return handlerList;
066    }
067
068    /**
069     * {@inheritDoc}
070     */
071    @Override
072    public void unsubscribeAll(@NotNull Key key) {
073        Objects.requireNonNull(key);
074
075        handlers.values().forEach(handlerList -> handlerList.unsubscribeAll(key));
076    }
077
078    /**
079     * {@inheritDoc}
080     */
081    @Override
082    @NotNull
083    @SuppressWarnings({"unchecked", "rawtypes"})
084    public <T extends Event> T callEvent(@NotNull T event) {
085        Objects.requireNonNull(event);
086
087        Class<T> eventClass = (Class<T>) event.getClass();
088        HandlerList<T> handlerList = searchHandlerList(eventClass);
089
090        if (handlerList != null) {
091            handlerList.post(event);
092        }
093
094        for (Class clazz = eventClass.getSuperclass();
095             clazz != null && Event.class.isAssignableFrom(clazz);
096             clazz = clazz.getSuperclass()) {
097
098            HandlerList subHandlerList = searchHandlerList((Class<? extends Event>) clazz);
099
100            if (subHandlerList != null) {
101                subHandlerList.post(event);
102            }
103        }
104
105        return event;
106    }
107
108    /**
109     * {@inheritDoc}
110     */
111    @SuppressWarnings("unchecked")
112    @Nullable
113    private <T extends Event> HandlerList<T> searchHandlerList(@NotNull Class<T> eventClass) {
114        return (HandlerList<T>) handlers.get(eventClass);
115    }
116
117    @Override
118    public String toString() {
119        return "SimpleEventBus{" +
120                "handlers=" + handlers +
121                '}';
122    }
123}
