025package com.github.siroshun09.event4j.event;
026
027/**
028 * An interface that indicates the cancel status of the event.
029 */
030public interface Cancellable {
031
032    /**
033     * Gets if the event has been cancelled.
034     *
035     * @return {@code true} if canceled, {@code false} otherwise.
036     */
037    boolean isCancelled();
038
039    /**
040     * Sets the cancel status of the event.
041     *
042     * @param cancel {@code true} to cancel, {@code false} to not.
043     */
044    void setCancelled(boolean cancel);
045}
